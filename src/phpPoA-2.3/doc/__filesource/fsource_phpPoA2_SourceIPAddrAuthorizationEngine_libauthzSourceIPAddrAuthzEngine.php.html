  26. /**
  27.  * This hook is executed right after retrieving source address and
  28.  * the arrays of allowed and denied patterns that will be checked inmediately.
  29.  * It can be used to alter the source address, and also to configure the filters on runtime.
  30.  * The hook receives the source IP address, the allowed and the denied patterns.
  31.  * Functions for this hook must be defined like this:
  32.  *
  33.  * function ipBeforeFilterHook(&$ipaddr, &$allowed, &$denied);
  34.  *
  35.  * Please bear in mind that hooks must return TRUE or they'll keep other hooks from executing.
  36.  */
  37. define("SOURCEADDR_BEFORE_FILTERS", "SOURCEADDR_BEFORE_FILTERS");
  38.  
  39. /**
  40.  * Authorization engine that works by checking the source IP address of the request.
  41.  * PLEASE NOTE THAT THIS ENGINE SILENTLY IGNORES BOTH USER AND ATTRIBUTES.
  42.  * PLEASE NOTE THAT THIS ENGINE WORKS ONLY FOR WEB-BASED APPLICATIONS.
  43.  * @package phpPoA2
  44.  * @subpackage SourceIPAddrAuthorizationEngine
  45.  */
  46. class SourceIPAddrAuthzEngine extends AuthorizationEngine {
  47.  
  48.     protected $valid_hooks = array(SOURCEADDR_BEFORE_FILTERS);
  49.  
  50.     /**
  51.      * PLEASE NOTE THAT THIS ENGINE SILENTLY IGNORES BOTH USER AND ATTRIBUTES.
  52.      */
  53.     public function isAuthorized($user, $attrs) {
  54.         $default = $this->cfg->getDefaultBehaviour();
  55.  
  56.         // proxy support
  57.         $src_addr = (isset($_SERVER['HTTP_X_FORWARDED_FOR'])) ? $_SERVER['HTTP_X_FORWARDED_FOR'] : $_SERVER['REMOTE_ADDR'];
  58.  
  59.         // check if there are IP filters
  60.         $allowed = $this->cfg->getAllowed();
  61.         $denied = $this->cfg->getDenied();
  62.  
  63.         // run hook before checking patterns
  64.         $args = array($src_addr, $allowed, $denied);
  65.         $this->runHooks(SOURCEADDR_BEFORE_FILTERS, $args);
  66.         $src_addr = $args[0];
  67.         $allowed = $args[1];
  68.         $denied = $args[2];
  69.  
  70.         $allowed_match = $this->matches($src_addr, $allowed);
  71.         $denied_match  = $this->matches($src_addr, $denied);
  72.  
  73.         // check matches giving priority to the default setting
  74.         $order = array($default, !$default);
  75.         foreach ($order as $option) {
  76.             if ($option) { // check allowed attributes
  77.                 if ($allowed_match) {
  78.                     trigger_error(PoAUtils::msg('source-ip-allowed', array($src_addr, $allowed_match)), E_USER_WARNING);
  79.                     return true;
  80.                 }
  81.             } else { // check denied attributes
  82.                 if ($denied_match) {
  83.                     trigger_error(PoAUtils::msg('source-ip-denied', array($src_addr, $denied_match)), E_USER_WARNING);
  84.                     return false;
  85.                 }
  86.             }
  87.         }
  88.  
  89.     // default response
  90.     trigger_error(PoAUtils::msg('authz-default-fallback'), E_USER_NOTICE);
  91.         return $default;
  92.     }

  97.         $this->registerHandler();
  98.         $list = $this->cfg->getAllowed();

  100.         return $list;
  101.     }

  104.         return false;

  111.     /**
  112.      * Check if an IP address matches the current allowed patterns.
  113.      * @param ip The IP address.
  114.      * @param patterns An array of patterns to be matched with.
  115.      * @return The matched pattern. False otherwise.
  116.      */
  117.     private function matches($addr, $patterns) {
  118.         // setup filtering criteria
  119.         $search = array("/\./",
  120.                         "/\.0/",
  121.                         // IPv6 support
  122.                         "/(:0){1,7}/",
  123.                         "/^::/",
  124.                         "/::$/");
  125.         $replace = array("\.",
  126.                          ".\d{1,3}",
  127.                          // IPv6 support
  128.                          "::",
  129.                          "(([0-9a-fA-F]{1,4})){1,7}\:",
  130.                          "(\:([0-9a-fA-F]{1,4})){1,7}");
  131.  
  132.         foreach ($patterns as $pattern) {
  133.             if (is_array($pattern)) continue; // arrays are not supported, just single strings
  134.  
  135.             // transform from network notation to regular expression
  136.             $mask = preg_replace($search, $replace, $pattern);
  137.  
  138.             if (preg_match("/".$mask."/i", $addr)) {
  139.                 return $pattern;
  140.             }
  141.         }
  142.         return false;
  143.     }
  144.  
  145. }
