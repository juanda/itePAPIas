  36. /**
  37.  * This hook is executed right before it is checked whether the user was invited or not to
  38.  * continue with authorization.
  39.  * It can be used to trick the engine to believe the user was previously invited (and therefore
  40.  * forcing authorization).
  41.  * The hook receives a boolean parameter representing whether the referenced invitation was found
  42.  * or not.
  43.  *
  44.  * function invitesBeforeAuthorization(&$invite_exists);
  45.  *
  46.  * Please bear in mind that hooks must return TRUE or they'll keep other hooks from executing.
  47.  */
  48. define("INVITES_BEFORE_AUTHORIZATION", "INVITES_BEFORE_AUTHORIZATION");

  82.     /**
  83.      * Check if the specified user is authorized with the given attributes.
  84.      * @param user The name of the user to check authorization for.
  85.      * @param attrs The attributes of the user to match his identity.
  86.      * @return boolean true if the user is authorized, false otherwise.
  87.      */
  88.     public function isAuthorized($user, $attrs) {
  89.         if (!$this->authz_db->open()) {
  90.             trigger_error(PoAUtils::msg('cannot-open-authz-db'), E_USER_WARNING);
  91.             return false;
  92.         }
  93.  
  94.         // retrieve data
  95.         $stored = $this->authz_db->fetch($user);
  96.         $this->authz_db->close();
  97.         if (!$stored) { // not found
  98.             trigger_error(PoAUtils::msg('cannot-fetch-key', array($user)), E_USER_WARNING);
  99.             return false;
  100.         }
  101.  
  102.         // check expiration
  103.         $now = time();
  104.         if ($stored['expires'] && $now > $stored['expires']) {
  105.             trigger_error(PoAUtils::msg('authz-expired', array($user)), E_USER_WARNING);
  106.             return false;
  107.         }
  108.  
  109.         // check attributes
  110.         $rslt = false;
  111.         foreach ($this->cfg->getUniqueAttributes() as $attr) {
  112.             if (is_array($attr)) {
  113.                 $partial = true;
  114.                 foreach ($attr as $compound) {
  115.                     $partial &= isset($stored['attributes'][$compound]) && // attribute is stored
  116.                                 isset($attrs[$compound]) && // attribute is set
  117.                                 // attribute IS NOT an array and the value is stored
  118.                                 ((!is_array($attrs[$compound]) && $stored['attributes'][$compound] === $attrs[$compound]) ||
  119.                                 // attribute IS an array and the value stored is in it
  120.                                 (is_array($attrs[$compound]) && in_array($stored['attributes'][$compound], $attrs[$compound])));
  121.                 }
  122.                 $rslt |= $partial;
  123.             } else {
  124.                 $rslt |= isset($stored['attributes'][$attr]) && // attribute is stored
  125.                          isset($attrs[$attr]) && // attribute is set
  126.                          // attribute IS NOT an array and the value is stored
  127.                          ((!is_array($attrs[$attr]) && $stored['attributes'][$attr] === $attrs[$attr]) ||
  128.                          // attribute IS an array and the value stored is in it
  129.                          (is_array($attrs[$attr]) && in_array($stored['attributes'][$attr], $attrs[$attr])));
  130.             }
  131.             if ($rslt) break;
  132.         }
  133.         return $rslt;
  134.     }

  159.         $this->registerHandler();
  160.         if (!$this->invites_db->open()) {
  161.             trigger_error(PoAUtils::msg('cannot-open-inv-db'), E_USER_WARNING);
  162.             return false;

  164.  
  165.         $all = $this->invites_db->fetch_all();

  167.         return $this->finish($all);

  170.     /**
  171.      * Authorize the specified user.
  172.      * @param user 
  173.      * @param attrs 
  174.      * @param ref 
  175.      * @param expires 
  176.      * @return boolean true if the user was successfully authorized, false otherwise.
  177.      */
  178.     public function authorize($user, $attrs, $ref, $expires = 0) {
  179.         if (!$this->authz_db->open()) {
  180.             trigger_error(PoAUtils::msg('cannot-open-authz-db'), E_USER_WARNING);
  181.             return false;
  182.         }
  183.  
  184.         if (!$this->invites_db->open()) {
  185.             trigger_error(PoAUtils::msg('cannot-open-inv-db'), E_USER_WARNING);
  186.             return false;
  187.         }
  188.  
  189.         $invited = $this->invites_db->check($ref);
  190.         $exists = $this->authz_db->check($user);
  191.  
  192.         // run hook before actually performing authorization
  193.         $args = array($invited);
  194.         $this->runHooks(INVITES_BEFORE_AUTHORIZATION, $args);
  195.         $invited = $args[0];
  196.  
  197.         // the user wasn't previously invited
  198.         if (!$invited) {
  199.             if ($exists) { // the user was previously authorized
  200.                 $stored = $this->authz_db->fetch($user);
  201.                 $matches = true;
  202.                 foreach ($stored['attributes'] as $name => $value) {
  203.                     $matches &= ($attrs[$name] == $value);
  204.                 }
  205.                 if ($matches) { // is the same user, skip
  206.                     trigger_error(PoAUtils::msg('user-already-authz', array($user)), E_USER_WARNING);
  207.                     return $this->finish(false);
  208.                 }
  209.             }
  210.             trigger_error(PoAUtils::msg('invite-non-existant', array($ref)), E_USER_WARNING);
  211.             return $this->finish(false);
  212.         }
  213.  
  214.         // either the invite exists or we are asked to force authorization
  215.         $invite = $this->invites_db->fetch($ref);
  216.  
  217.         // check if the user has some of the mandatory attributes
  218.         $unique = $this->cfg->getUniqueAttributes();
  219.         foreach ($unique as $item) {
  220.             $save = array(); // restore on each iteration
  221.             if (is_array($item)) { // a combination of attributes
  222.                 $complete = true;
  223.                 foreach ($item as $name) {
  224.                     $complete &= !empty($attrs[$name]);
  225.                     $value = $attrs[$name];
  226.                     // multiple values?
  227.                     if (is_array($attrs[$name])) {
  228.                         $value = $attrs[$name][0];
  229.                     }
  230.                     $save[$name] = $value;
  231.                 }
  232.                 if ($complete) {
  233.                     break; // all attributes found, ok!
  234.                 }
  235.             } else { // a single attribute
  236.                 if (!empty($attrs[$item])) { // attribute found, ok!
  237.                     $value = $attrs[$item];
  238.                     // multiple values?
  239.                     if (is_array($attrs[$item])) {
  240.                         $value = $attrs[$item][0];
  241.                     }
  242.                     $save[$item] = $value;
  243.                     break;
  244.                 }
  245.             }
  246.         }
  247.         if (empty($save)) { // no available attributes!
  248.             trigger_error(PoAUtils::msg('missing-attrs'), E_USER_WARNING);
  249.             return $this->finish(false);
  250.         }
  251.  
  252.         // e-mail verification
  253.         if (!$this->emailVerify($invite['email'], $attrs)) {
  254.              return $this->finish(false);
  255.         }
  256.  
  257.         // now save the user in the authorized database
  258.         if (@!$this->authz_db->replace_authorization($user, $save, $invite['email'], $expires)) {
  259.             trigger_error(PoAUtils::msg('cannot-authorize', array($user)), E_USER_WARNING);
  260.             return $this->finish(false);
  261.         }
  262.  
  263.         // remove invite
  264.         if (@!$this->invites_db->delete($ref)) {
  265.             trigger_error(PoAUtils::msg('cannot-del-invite', array($user, $ref)), E_USER_WARNING);
  266.             return $this->finish(false);
  267.         }
  268.  
  269.         // look for any other pending invites for this user, and delete them
  270.         $pending = $this->invites_db->fetch_all();
  271.         foreach ($pending as $key => $stored) {
  272.             if ($stored['email'] === $invite['email']) {
  273.                 // clean up
  274.                 if (@!$this->invites_db->delete($key)) {
  275.                     trigger_error(PoAUtils::msg('cannot-del-invite', array($user, $key)), E_USER_WARNING);
  276.                     return $this->finish(false);
  277.                 }
  278.             }
  279.         }
  280.  
  281.         // success
  282.         trigger_error(PoAUtils::msg('user-authorized', array($user)), E_USER_NOTICE);
  283.         return $this->finish(true);
  284.     }
  285.  
  286.     /**
  287.      * Revoke authorization for the specified user.
  288.      * @param 
  289.      * @return boolean true if authorization was successfully revoked, false otherwise.
  290.      */
  291.     public function revoke($mail) {
  292.         if (!$this->authz_db->open()) {
  293.             trigger_error(PoAUtils::msg('cannot-open-authz-db'), E_USER_WARNING);
  294.             return false;
  295.         }
  296.  
  297.         // JPC20110124
  298.         // Library should leave the address as is.
  299.         // sanitize email
  300.         //$mail = strtolower($mail);
  301.  
  302.         // get a list with all users authorized 
  303.         $all = $this->authz_db->fetch_all();
  304.  
  305.         // iterate to find the user
  306.         foreach ($all as $key => $values) {
  307.             if ($values['email'] == $mail) {
  308.                 // found, remove!
  309.                 return $this->finish($this->authz_db->delete($key));
  310.             }
  311.         }
  312.  
  313.         // not found!
  314.         return $this->finish(false);
  315.     }
  316.  
  317.     /**
  318.      * Send an invitation to an e-mail address (that is, send an e-mail to that
  319.      * address with instructions on how to get authorized and an URL to follow).
  320.      * @param mail The e-mail of the user.
  321.      * @param expires The time (POSIX) when authorization will expire. Use 0 if authorization
  322.      *  should never expire. Defaults to 0.
  323.      * @return boolean true if the invitation was correctly sent, false in any other case.
  324.      */
  325.     public function invite($mail, $expires = 0) {
  326.         $this->registerHandler();
  327.         if (!$this->invites_db->open()) {
  328.             trigger_error(PoAUtils::msg('cannot-open-inv-db'), E_USER_WARNING);
  329.             return false;
  330.         }
  331.  
  332.         // JPC20110124
  333.         // Library should leave the address as is.
  334.         // sanitize e-mail
  335.         //$mail = strtolower($mail);
  336.  
  337.         // generate random reference
  338.         $ref = mt_rand();
  339.  
  340.         if (@!$this->invites_db->replace_invite($ref, $mail, $expires)) {
  341.             trigger_error(PoAUtils::msg('cannot-save-invite', array($mail)), E_USER_WARNING);
  342.             return $this->finish(false);
  343.         }
  344.  
  345.         // setup email
  346.         $sep = (!strstr($this->cfg->getInviteURL(), "?")) ? "?" : "&";
  347.         $url = $this->cfg->getInviteURL().$sep."ref=".$ref;
  348.         $text = preg_replace("/##URL##/", $url, $this->cfg->getInviteText());
  349.         $headers = "From: ".$this->cfg->getAdminEmail();
  350.         $command_params = "-f ".$this->cfg->getAdminEmail();
  351.  
  352.         // send it
  353.         mail($mail, $this->cfg->getInviteSubject(), $text, $headers, $command_params);
  354.  
  355.         trigger_error(PoAUtils::msg('invite-sent-to', array($mail)), E_USER_WARNING);
  356.         $this->clean();
  357.         return $this->finish(true);
  358.     }
  359.  
  360.     /**
  361.      * Remove an invitation from the database.
  362.      * @param ref The reference to the invite to remove.
  363.      * @return boolean true if the invite was removed, false otherwise.
  364.      */
  365.     public function removeInvite($ref) {
  366.         $this->registerHandler();
  367.         if (!$this->invites_db->open()) {
  368.             trigger_error(PoAUtils::msg('cannot-open-inv-db'), E_USER_WARNING);
  369.             return false;
  370.         }
  371.  
  372.         // remove the invite
  373.         $rslt = $this->invites_db->delete($ref);
  374.         trigger_error(PoAUtils::msg('invite-removed', array($ref)), E_USER_WARNING);
  375.         $this->clean();
  376.         return $this->finish($rslt);
  377.     }
  378.  
  379.     /**
  380.      * Perform e-mail verification for the current user according to the configuration
  381.      * for this site.
  382.      * @param mail The e-mail of the user.
  383.      * @param attrs The array of attributes of the user.
  384.      * @return boolean true if verification succeeds or was not performed, false otherwise.
  385.      */
  386.     protected function emailVerify($mail, $attrs) {
  387.         $mail_attr = $this->cfg->getEmailVerifyAttribute();
  388.         if ($this->cfg->doEmailVerify()) {
  389.             // empty attribute, configuration error!
  390.             if (empty($mail_attr)) {
  391.                 trigger_error(PoAUtils::msg('mail-attr-err'), E_USER_WARNING);
  392.                 return false;
  393.             }
  394.             // attribute not set, cannot verify
  395.             if (!isset($attrs[$mail_attr])) {
  396.                 trigger_error(PoAUtils::msg('missing-mail-attr', array($mail_attr)), E_USER_WARNING);
  397.                 return false;
  398.             }
  399.  
  400.             $alg_re = $this->cfg->getEmailVerifyAlgRegEx();
  401.             $val_re = $this->cfg->getEmailVerifyRegEx();
  402.  
  403.             // support for attributes with multiple values
  404.             $mail_attrs = $attrs[$mail_attr];
  405.             if (!is_array($attrs[$mail_attr])) {
  406.                 $mail_attrs = array($attrs[$mail_attr]);
  407.             }
  408.  
  409.             $match = false;
  410.             foreach ($mail_attrs as $attr_val) {
