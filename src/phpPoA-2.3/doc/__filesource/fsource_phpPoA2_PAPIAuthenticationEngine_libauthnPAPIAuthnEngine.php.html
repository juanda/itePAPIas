  168.     public function authenticate() {
  169.         // PAPI authentication protocol v1.0
  170.  
  171.         $action = @array_key_exists('ACTION', $_REQUEST) ? $_REQUEST['ACTION'] : "";
  172.         $auth = (isset($_COOKIE[$this->cookie_name])) ? $this->testCookie() : false;
  173.  
  174.         // check if we have a cookie or coming back from AS/GPoA
  175.         if ($action === "CHECKED" && !$auth) { // GPoA/AS response
  176.             $data = $_REQUEST['DATA'];
  177.             $key = $this->testResponse($data, $this->pkey);
  178.             if (!$key) {
  179.                 $this->status = AUTHN_FAILED;
  180.                 $this->dirty = false;
  181.                 return AUTHN_FAILED;
  182.             }
  183.             $request = $this->loadRequest($key);
  184.             if (!$request) {
  185.                 trigger_error(PoAUtils::msg('unknown-request', array()), E_USER_WARNING);
  186.                 $this->status = AUTHN_FAILED;
  187.                 $this->dirty = false;
  188.                 return AUTHN_FAILED;
  189.             }
  190.             $this->deleteRequest($key);
  191.             if ($key) { // the request was validated
  192.                 if ($this->skip_redirection) {
  193.                     $this->status = AUTHN_SUCCESS;
  194.                     return AUTHN_SUCCESS;
  195.                 }
  196.  
  197.                 // set a new cookie
  198.                 $c = $this->getNewCookie($this->assertion);
  199.                 if (setcookie($this->cookie_name, $c, 0, $this->cfg->getLocation(), $this->cfg->getCookieDomain(), 0)) {
  200.                     $_COOKIE[$this->cookie_name] = $c;
  201.  
  202.                     // run hooks
  203.                     $arg = array($this->clean_url);
  204.                     $this->runHooks(PAPI_CLEAN_URL, $arg);
  205.                     $this->clean_url = $arg[0];
  206.  
  207.                     // finish it off
  208.                     if (!$this->clean_url) {
  209.                         $this->status = AUTHN_SUCCESS;
  210.                         return AUTHN_SUCCESS;
  211.                     } else {
  212.                         if ($_SERVER["REQUEST_METHOD"] === "POST") {
  213.                             $inputs = "";
  214.  
  215.                             // build HTML with an input element for each element in the query string
  216.                             parse_str($_SERVER["QUERY_STRING"], $params);
  217.                             foreach ($params as $name => $value) {
  218.                                 $inputs .= "<input name='".$name."' type='hidden' value='".$value."' />";
  219.                             }
  220.                             $inputs .= "<input type='submit' value='".PoAUtils::msg('continue', array())."' />";
  221.  
  222.                             // print a HTML form to continue
  223. ?>
  224. <html>
  225.  <head>
  226.   <title>phpPoA2 transaction in progress...</title>
  227.  </head>
  228.  <body onload="document.forms[0].submit();">
  229.   <form action="<?php echo $_SERVER['SCRIPT_NAME'];?>" method="post">
  230.    <?php echo $inputs; ?>
  231.   </form>
  232.  </body>
  233. </html>
  234. <?php
  235.                             die(0);
  236.                         } else {
  237.                             // build redirect URL
  238.                             $protocol = (!empty($_SERVER['HTTPS'])) ? "https://" : "http://";
  239.                             $url = $protocol.$_SERVER['SERVER_NAME'].":".$_SERVER['SERVER_PORT'];
  240.                             $url .= substr($_SERVER['REQUEST_URI'], 0, strpos($_SERVER['REQUEST_URI'], "ACTION=CHECKED") -1);
  241.                             $this->redirect($url);
  242.                         }
  243.                     }
  244.                 } else { // can't set the cookie
  245.                     trigger_error(PoAUtils::msg('cannot-set-cookie', array()), E_USER_WARNING);
  246.                 }
  247.             } else { // the request is invalid!
  248.                 trigger_error(PoAUtils::msg('authn-err', array()), E_USER_WARNING);
  249.             }
  250.             $this->status = AUTHN_FAILED;
  251.             return AUTHN_FAILED;
  252.         } else if (!$auth) { // first time browser access (w/o cookie)
  253.             if ($this->skip_redirection) {
  254.                 $this->status = AUTHN_FAILED;
  255.                 throw new PoAException('cookie-not-found', E_USER_ERROR, array($this->cookie_name));
  256.             }
  257.             trigger_error(PoAUtils::msg('cookie-not-found', array($this->cookie_name)), E_USER_WARNING);
  258.             $this->deleteCookie();
  259.             $this->redirect();
  260.         } else { // valid access with cookie, update it!
  261.             $c = $this->getNewCookie($this->assertion);
  262.             if (setcookie($this->cookie_name, $c, 0, $this->cfg->getLocation(), $this->cfg->getCookieDomain(), 0)) {
  263.                 $_COOKIE[$this->cookie_name] = $c;
  264.                 // if no request found, assume this is a visit to a previously stored URL
  265.                 // (reloaded by the user or in the browser favorites/history).
  266.                 // Rebuild the request without the ACTION and DATA parameters
  267.  
  268.                 // set protocol
  269.                 $protocol = (!empty($_SERVER['HTTPS'])) ? "https://" : "http://";                
  270.  
  271.                 // rebuild location
  272.                 $re[] = "/ACTION=[^&]*&?/";
  273.                 $re[] = "/DATA=[^&]*&?/";
  274.                 $location = $protocol.$_SERVER["SERVER_NAME"].":".$_SERVER["SERVER_PORT"].
  275.                             rtrim(preg_replace($re,"",$_SERVER["REQUEST_URI"]),"&?");
  276.  
  277.                 // run hooks
  278.                 $arg = array($this->clean_url);
  279.                 $this->runHooks(PAPI_CLEAN_URL, $arg);
  280.                 $this->clean_url = $arg[0];
  281.  
  282.                 if ((isset($_REQUEST["ACTION"]) || isset ($_REQUEST["DATA"])) && !$this->skip_redirection && $this->clean_url) {
  283.                     $this->redirect($location);
  284.                 }
  285.  
  286.             } else { // set cookie failed!
  287.                 trigger_error(PoAUtils::msg('cannot-set-cookie', array()), E_USER_WARNING);
  288.             }
  289.  
  290.             $this->status = AUTHN_SUCCESS;
  291.             return AUTHN_SUCCESS;
  292.         }
  293.  
  294.         $this->status = AUTHN_FAILED;
  295.         return AUTHN_FAILED;
  296.     }

  308.     public function getAttributes() {
  309.         // avoid parsing the assertion again and again
  310.         if (empty($this->attributes)) {
  311.             $attrs = explode(ATTR_SEPARATOR, $this->assertion);
  312.             foreach ($attrs as $attr) {
  313.                 @list($name, $value) = explode(NAMEVALUE_SEPARATOR, $attr);
  314.  
  315.                 // discard operational attributes
  316.                 if (strpos($name, PROTO_ATTR_PREFIX) == 1) continue;
  317.  
  318.                 if (!empty($value)) {
  319.                     $values = explode(VALUE_SEPARATOR, $value);
  320.                     if (count($values) > 1) {
  321.                         $this->attributes[$name] = $values;
  322.                     } else {
  323.                         $this->attributes[$name] = $value;
  324.                     }
  325.                 }
  326.             }
  327.         }
  328.  
  329.         $arg = array($this->assertion, $this->attributes);
  330.         $this->runHooks(PAPI_ATTRIBUTE_PARSER, $arg);
  331.         $this->attributes = $arg[1];
  332.  
  333.         return $this->attributes;
  334.     }

  336.     public function getAttribute($name, $namespace = NS_PAPI_ATTRIBUTES) {
  337.         switch ($namespace) {
  338.             case NS_PAPI_ATTRIBUTES:
  339.                 if (empty($this->attributes)) {
  340.                     $this->getAttributes();
  341.                 }
  342.                 $attr = $this->attributes[$name];
  343.                 if (!isset($this->attributes[$name])) {
  344.                     // attribute query
  345.                     $attr = $this->attributeQuery($name);
  346.                 }
  347.                 break;
  348.             case NS_PAPI_PROTOCOL:
  349.                 switch ($name) {
  350.                     case PROTO_ATTR_AS_ID:
  351.                         $attr = $this->as_id;
  352.                         break;
  353.                     case PROTO_ATTR_KEY:
  354.                         $attr = $this->key;
  355.                         break;
  356.                     case PROTO_ATTR_EXPIRE_TIME:
  357.                         $attr = $this->expiration_time;
  358.                         break;
  359.                     default:
  360.                         $attr = @$this->attributes['_papi_'.$name];
  361.                 }
  362.         }
  363.         return $attr;
  364.     }

  366.     public function logout($slo = false) {
  367.         // first check if we really need to logout!
  368.         if (!$this->isAuthenticated()) {
  369.             trigger_error(PoAUtils::msg('already-logged-out', array()), E_USER_NOTICE);
  370.             return true;
  371.         }
  372.  
  373.         // local logout only
  374.         if (!$slo) {
  375.             $this->deleteCookie();
  376.             trigger_error(PoAUtils::msg('local-logout-success', array()), E_USER_NOTICE);
  377.             return true;
  378.         }
  379.  
  380.         // check configuration
  381.         $rtype = $this->cfg->getRedirectType();
  382.         if ($rtype === AS_T) {
  383.             // configuration error, redirection type must be GPOA_T and the
  384.             throw new PoAException('slo-conf-error', E_USER_ERROR, array());
  385.         }
  386.  
  387.         // single logout
  388.         $action = (array_key_exists('ACTION', $_REQUEST)) ? $_REQUEST['ACTION'] : "";
  389.         if ($action === "PAPILOGOUT") { // single logout
  390.             // GPoA asks for logout!
  391.             $this->deleteCookie();
  392.             trigger_error(PoAUtils::msg('slo-logout', array()), E_USER_NOTICE);
  393.             $location = $this->getSingleLogoutResponseLocation();
  394.         } else { // logout is triggered from the application
  395.             trigger_error(PoAUtils::msg('slo-requested', array()), E_USER_NOTICE);
  396.             $location = $this->getSingleLogoutLocation();
  397.         }
  398.         $this->redirect($location);
  399.     }

  408.     /**
  409.      * Check if a cookie is valid.
  410.      * @param cookie The cookie.
  411.      * @return boolean true if the cookie is valid, false otherwise.

  413.     protected function testCookie($name = "") {
  414.         if (empty($name)) {
  415.             $name = $this->cookie_name;

  419.         if (empty($cookie)) {

  422.         }
  423.  
  424.         $now = time();

  426.         $newsource = $this->crypto->decryptAES($cookie);
  427.         list($timestamp, $this->global_expire_time, $location, $id, $this->as_id, $this->assertion) = explode(":", $newsource, 6);
  428.  
  429.         $this->expiration_time = ($this->global_expire_time < ($timestamp + $this->cfg->getCookieTimeout()))
  430.                                  ? $this->global_expire_time : $timestamp + $this->cfg->getCookieTimeout();

  442.         }
  443.         if (($this->global_expire_time < $now) or ($timestamp + $this->cfg->getCookieTimeout() < $now)) {

  445.

  447.         }

  456.     protected function deleteCookie() {
  457.         /*
  458.          * This is a hack to make cookie deletion work with firefox browers.
  459.          * Firefox won't delete a cookie unless it is set with exactly the same
  460.          * way it was originally set, except the expiration time. So we have to

  463.          */

  465.                   @$_COOKIE[$this->cookie_name],
  466.                   time()-3600,

  468.                   $this->cfg->getCookieDomain(), 0);
  469.         unset($_COOKIE[$this->cookie_name]);

  472.     /**
  473.      * Check the response from the AS/GPoA.
  474.      * @param data The data received.
  475.      * @param key The public key of the AS/GPoA.
  476.      * @return boolean true if valid, false else.
  477.      */
  478.     protected function testResponse($data, $pubkey) {
  479.         // decrypt data
  480.         $newsource = $this->crypto->decrypt($data);
  481.  
  482.         if ($newsource === false) {
  483.             // Cannot decrypt!
  484.             trigger_error(PoAUtils::msg('cannot-decrypt', array()), E_USER_WARNING);
  485.             return false;
  486.         }
  487.  
  488.         // check the assertion
  489.         if (empty($newsource)) { // empty assertion
  490.             trigger_error(PoAUtils::msg('empty-response-err', array()), E_USER_WARNING);
  491.             return false;
  492.         }
  493.  
  494.         // parse data
  495.         $response = explode(":", $newsource);
  496.         $this->key = array_pop($response);
  497.         $current_time = array_pop($response);
  498.         $this->global_expire_time = array_pop($response);
  499.         $rest = implode(":", $response);
  500.         $rest_a = explode("@", $rest);
  501.         $this->as_id = array_pop($rest_a);
  502.         $this->assertion = implode("@", $rest_a);
  503.  
  504.         $this->expiration_time = ($this->global_expire_time < ($current_time + $this->cfg->getCookieTimeout()))
  505.                                  ? $this->global_expire_time : $current_time + $this->cfg->getCookieTimeout();
  506.  
  507.  
  508.         if ($this->assertion === "ERROR") { // AS/GPoA error response, authentication failed!
  509.             $this->deleteRequest($this->key);
  510.             trigger_error(PoAUtils::msg('authn-error', array()), E_USER_WARNING);
  511.             return false;
  512.         }
  513.  
  514.         // check timestamps
  515.         if ($this->global_expire_time < time()) { // globally expired
  516.             $this->deleteRequest($this->key);
  517.             trigger_error(PoAUtils::msg('expired-response', array()), E_USER_WARNING);
  518.             return false;
  519.         }
  520.         if ($current_time + $this->cfg->getCookieTimeout() < time()) { // expired
  521.             $this->deleteRequest($this->key);
  522.             trigger_error(PoAUtils::msg('expired-response', array()), E_USER_WARNING);
  523.             return false;
  524.         }
  525.  
  526.         // the response is OK
  527.         trigger_error(PoAUtils::msg('valid-response', array($this->assertion)), E_USER_WARNING);
  528.         return $this->key;
  529.     }
