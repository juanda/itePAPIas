  26. /**
  27.  * This hook is executed right after retrieving the current URI, the params (both GET and POST) and
  28.  * the arrays of allowed and denied patterns that will be checked inmediately.
  29.  * It can be used to alter parameters and the URL, and also to configure the filters on runtime.
  30.  * The hook receives the URI string, an array of parameters, the allowed and the denied patterns.
  31.  * Functions for this hook must be defined like this:
  32.  *
  33.  * function queryBeforeFilterHook(&$uri, &$params, &$allowed, &$denied);
  34.  *
  35.  * Please bear in mind that hooks must return TRUE or they'll keep other hooks from executing.
  36.  */
  37. define("QUERY_BEFORE_FILTERS", "QUERY_BEFORE_FILTERS");

  39. /**
  40.  * Authorization engine that works by checking the query string of the request.
  41.  * PLEASE NOTE THAT THIS ENGINE SILENTLY IGNORES BOTH USER AND ATTRIBUTES.
  42.  * PLEASE NOTE THAT THIS ENGINE WORKS ONLY FOR WEB-BASED APPLICATIONS.
  43.  * @package phpPoA2
  44.  * @subpackage QueryFilterAuthorizationEngine
  45.  */
  46. class QueryFilterAuthzEngine extends AuthorizationEngine {
  47.  
  48.     protected $valid_hooks = array(QUERY_BEFORE_FILTERS);
  49.  
  50.     /**
  51.      * PLEASE NOTE THAT THIS ENGINE SILENTLY IGNORES BOTH USER AND ATTRIBUTES.
  52.      */
  53.     public function isAuthorized($user, $attrs) {
  54.         $params = $this->getQueryParams();
  55.         $default = $this->cfg->getDefaultBehaviour();
  56.         $allowed = $this->cfg->getAllowedPatterns();
  57.         $denied  = $this->cfg->getDeniedPatterns();
  58.  
  59.         // run hook before checking patterns
  60.         $args = array($params, $allowed, $denied);
  61.         $this->runHooks(QUERY_BEFORE_FILTERS, $args);
  62.         $params = $args[0];
  63.         $allowed = $args[1];
  64.         $denied = $args[2];
  65.  
  66.         $allowed_match = $this->matches($params, $allowed);
  67.         $denied_match  = $this->matches($params, $denied);
  68.  
  69.         // check matches giving priority to the default setting
  70.         $order = array($default, !$default);
  71.         foreach ($order as $option) {
  72.             if ($option) { // check allowed parameters
  73.                 if ($allowed_match) {
  74.                     trigger_error(PoAUtils::msg('allowed-param-match', array($allowed_match)), E_USER_WARNING);
  75.                     return true;
  76.                 }
  77.             } else { // check denied parameters
  78.                 if ($denied_match) {
  79.                     trigger_error(PoAUtils::msg('denied-param-match', array($denied_match)), E_USER_WARNING);
  80.                     return false;
  81.                 }
  82.             }
  83.         }
  84.  
  85.         // default response
  86.         trigger_error(PoAUtils::msg('authz-default-fallback'), E_USER_NOTICE);
  87.         return $default;
  88.     }

  105.     /**
  106.      * Get all the input received for the current request.
  107.      * @return data An array containing all parameters received as input.
  108.      */
  109.     private function getQueryParams() {
  110.         // Full URI
  111.         $uri = $_SERVER['SERVER_NAME'].$_SERVER['REQUEST_URI'];
  112.  
  113.         // GET queries
  114.         $data = explode("&", $_SERVER['QUERY_STRING']);
  115.  
  116.         // POST queries
  117.         if ($_SERVER['REQUEST_METHOD'] === "POST") {
  118.             $str = file_get_contents(STDIN);
  119.  
  120.             $post = array();
  121.             if ($str) {
  122.                 $post = explode("&". $str);
  123.             }
  124.  
  125.             $data = array_merge($data, $post);
  126.         }
  127.  
  128.         // convert to associative array
  129.         $result['&REQUEST_URI'] = $uri;
  130.         foreach ($data as $item) {
  131.             list($key, $value) = explode("=", $item);
  132.             $result[$key] = $value;
  133.         }
  134.  
  135.         return $result;
  136.     }

  138.     /**
  139.      * Returns the parameter (or parameters) that matched a list of patterns.
  140.      * @param params An array of parameters to check.
  141.      * @param patterns An array of patterns to be matched with.
  142.      * @return The names of the parameters matched, comma separated if more than one.
  143.      *  False otherwise.
  144.      */
  145.     private function matches($params, $patterns) {
  146.         $match = false;
  147.         foreach ($patterns as $key => $value) {
  148.             if (is_numeric($key) && is_array($value)) { // must match a bunch of options
  149.                 $partial_match = true;
  150.                 $matches = array();
  151.                 foreach ($value as $name => $pattern) {
  152.                     if (!isset($params[$name])) { // parameter not set, skip this option
  153.                         $partial_match = false;
  154.                         break;
  155.                     }
  156.  
  157.                     // convert parameter to array for easy handling
  158.                     $param = $params[$name];
  159.                     if (!is_array($params[$name])) {
  160.                         $param = array($params[$name]);
  161.                     }
  162.  
  163.                     // convert pattern to array for easy handling
  164.                     $pats = $pattern;
  165.                     if (!is_array($pattern)) {
  166.                         $pats = array($pattern);
  167.                     }
  168.  
  169.                     // check if any of the possible values match
  170.                     $some_val_matches = false;
  171.                     foreach ($param as $item) {
  172.                         foreach ($pats as $pat) {
  173.                             if (preg_match('/'.$pat.'/', $item)) {
  174.                                 // parameter matches, continue
  175.                                 $some_val_matches = true;
  176.                                 break;
  177.                             }
  178.                         }
  179.                         if ($some_val_matches) break;
  180.                     }
  181.                     if (!$some_val_matches) {
  182.                         $partial_match = false;
  183.                         break;
  184.                     }
  185.  
  186.                     $matches[] = $name;
  187.                 }
  188.                 if ($partial_match) {
  189.                     $match = implode(",", $matches);
  190.                 }
  191.  
  192.             } else { // match just one parameter
  193.                 if (is_numeric($key))
  194.                     $key = "&REQUEST_URI"; // try to match the complete URI
  195.  
  196.                 if (!isset($params[$key])) // parameter not set, skip this option
  197.                     continue;
  198.  
  199.                 // convert parameter to array for easy handling
  200.                 $param = $params[$key];
  201.                 if (!is_array($params[$key])) {
  202.                     $param = array($params[$key]);
  203.                 }
  204.  
  205.                 // convert pattern to array for easy handling
  206.                 $pats = $value;
  207.                 if (!is_array($value)) {
  208.                     $pats = array($value);
  209.                 }
  210.  
  211.                 // check if any of the possible values match
  212.                 foreach ($param as $item) {
  213.                     foreach ($pats as $pattern) {
  214.                         if (preg_match('/'.$pattern.'/', $item)) {
  215.                             // parameter matches, stop searching
  216.                             $match = $key;
  217.                             break;
  218.                         }
  219.                     }
  220.                     if ($match) break;
  221.                 }
  222.             }
  223.         }
  224.         return $match;
  225.     }
  226.  
  227. }
